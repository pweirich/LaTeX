% These custom section headings below conform fairly closely
% to the Language guidelines.  They are created with the 
% titlesec package and they directly replace the usual 
% section headings, so \section, \subsection, and \subsubsection
% can be called as normal.

% According to the guidelines, only the first letter of the heading 
% should be capitalized.

% custom section
\titleformat{\section}[runin]
{\normalfont\scshape}
{\bfseries\thesection.}{.5em}{}[.]
\titlespacing{\section}
{\parindent}{*2}{\wordsep}

% custom subsection
\titleformat{\subsection}[runin]
{\normalfont\scshape}
{\bfseries\thesubsection.}{.5em}{}[.]
\titlespacing{\subsection}
{\parindent}{*2}{\wordsep}

% Language style guidelines call for no numbering
% beyond the second section level.
% custom subsubsection
\titleformat{\subsubsection}[runin]
{\normalfont\scshape}
{\bfseries}{1em}{}[.]
\titlespacing{\subsection}
{\parindent}{*2}{\wordsep}
